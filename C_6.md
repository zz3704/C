# 1. 구구단
```C
#include <stdio.h>

int main(void) {

  int i = 2, y = 1, ho;

  while (y < 11) {
    ho = i * y;
    printf("%d*%d=%d\n", i, y, ho);

    y++;
  }
}
```
# 2. 1부터 n까지의 합 
```C
#include <stdio.h>

int main(void) {

  int n, i = 1, hap = 0;

  printf("정수를 입력하시오 ");
  scanf("%d", &n);
  while (i <= n) {
    hap += i;
    i++;
  }
  printf("1부터 %d까지의 합은 %d입니다.", n, hap);
}
```
# 3. 1부터 n까지의 짝수합
```C
#include <stdio.h>

int main(void) {

  int n, i = 0, hap = 0;

  printf("정수를 입력하시오 ");
  scanf("%d", &n);

  while (i <= n) {
    hap += i;
    i = i + 2;
  }
  printf("1부터 %d까지의 짝수합은 %d입니다.", n, hap);
}
```
# 4. 스캔문 반복하여 총 합계구하기
```C
#include <stdio.h>

int main(void) {

  int i, y = 0, hap = 0;

  while (y < 5) {
    printf("값을 입력하시오 ");
    scanf("%d", &i);
    y++;
    hap += i;
  }
  printf("합계는 %d입니다.", hap);
}
```
# 5. do,while 이용해서 메뉴선택

```C 
#include <stdio.h>

int main(void) {

  int i = 0;
  do {
    printf("1---새로만들기\n");
    printf("2---파일열기\n");
    printf("3---파일닫기\n");
    printf("하나를 선택하시요.\n");
    scanf("%d", &i);
  } while (i < 1 || i > 3);
  printf("선택된 메뉴 = %d\n", i);
} 
```
