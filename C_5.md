
#include <stdio.h>

int main(void) {
  char ch;
  printf("문자를 입력하시오 : ");

  ch = getchar();
  if (ch >= 65 && ch <= 90) {
    printf("%c는 대문자입니다.\n", ch);
  } else if (ch >= 'a' && ch <= 'z') {
    printf("%c는 소문자입니다.\n", ch);
  } else if (ch >= '0' && ch <= '9') {
    printf("%c는 숫자입니다.\n", ch);
  } else {
    printf("%c는 기타문자입니다.\n", ch);
  }
}


#include <stdio.h>

int main(void) {
  int a, b;
  scanf("%d %d", &a, &b);

  if (a > b) {
    printf(">");
  } else if (a < b) {
    printf("<");
  } else if (a == b) {
    printf("==");
  }
}
