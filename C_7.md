# a,b값을 입력하면 더한값을 출력, 0일때 끝나는

 ```c 
#include <stdio.h>

int main(void) {
    int a,b;
    while (1) {
        scanf("%d %d", &a, &b);
        if(a==0 && b==0) break;
        printf("%d\n", a+b);
    }
}
 ```
 
 # 함수를 이용하여 프로그램 출력
 #include <stdio.h>

void print_stars() {
  for (int i = 0; i < 30; i++) {
    printf("*");
  }
}

int main(void) {
  print_stars();
  printf("\nHello World\n");
  print_stars();
  printf("\n");
}
